//STUDENT TABLE

//MySQL
CREATE TABLE STUDENT(
USN VARCHAR(10) PRIMARY KEY,
SNAME VARCHAR(25),
ADDRESS VARCHAR(25),
PHONE INTEGER,
GENDER VARCHAR(6));

//SQL Plus
CREATE TABLE STUDENT(
USN VARCHAR(10) PRIMARY KEY,
SNAME VARCHAR(25),
ADDRESS VARCHAR(25),
PHONE NUMBER(10),
GENDER VARCHAR(6));

// SEMSEC TABLE

//MySQL
CREATE TABLE SEMSEC(
SSID VARCHAR(10) PRIMARY KEY,
SEM INTEGER,
SEC VARCHAR(1));

//SQL Plus
CREATE TABLE SEMSEC(
SSID VARCHAR(10) PRIMARY KEY,
SEM NUMBER(2),
SEC VARCHAR(1));

// CLASS TABLE

//MYSQL
CREATE TABLE CLASS(
USN VARCHAR(10),
SSID VARCHAR(10),
PRIMARY KEY(USN,SSID),
FOREIGN KEY(USN) REFERENCES STUDENT(USN),
FOREIGN KEY(SSID) REFERENCES SEMSEC(SSID));

//SQL Plus
CREATE TABLE CLASS(
USN VARCHAR(10),
SSID VARCHAR(10),
PRIMARY KEY(USN,SSID),
FOREIGN KEY(USN) REFERENCES STUDENT(USN),
FOREIGN KEY(SSID) REFERENCES SEMSEC(SSID));

// COURSE TABLE

//MySQL
CREATE TABLE COURSE(
SUBCODE VARCHAR(10) PRIMARY KEY,
TITLE VARCHAR(25),
SEM INTEGER,
CREDITS INTEGER);

//SQL Plus
CREATE TABLE COURSE(
SUBCODE VARCHAR(10) PRIMARY KEY,
TITLE VARCHAR(25),
SEM NUMBER(2),
CREDITS NUMBER(2));

// IAMARKS TABLE

//MySQL
CREATE TABLE IAMARKS(
USN VARCHAR(10),
SUBCODE VARCHAR(10),
SSID VARCHAR(10),
TEST1 INTEGER,
TEST2 INTEGER,
TEST3 INTEGER,
FINALIA INTEGER,
PRIMARY KEY(SUBCODE,USN,SSID),
FOREIGN KEY(USN) REFERENCES STUDENT(USN),
FOREIGN KEY(SUBCODE) REFERENCES COURSE(SUBCODE),
FOREIGN KEY(SSID) REFERENCES SEMSEC(SSID));

//SQL Plus
CREATE TABLE IAMARKS(
USN VARCHAR(10),
SUBCODE VARCHAR(10),
SSID VARCHAR(10),
TEST1 NUMBER(2),
TEST2 NUMBER(2),
TEST3 NUMBER(2),
FINALIA NUMBER(2),
PRIMARY KEY(SUBCODE,USN,SSID),
FOREIGN KEY(USN) REFERENCES STUDENT(USN),
FOREIGN KEY(SUBCODE) REFERENCES SUBJECT(SUBCODE),
FOREIGN KEY(SSID) REFERENCES SEMSEC(SSID));

-- QUERY 1

SELECT S.*, SS.SEM, SS.SEC
FROM STUDENT S, SEMSEC SS, CLASS C
WHERE S.USN = C.USN AND
SS.SSID = C.SSID AND
SS.SEM = 4 AND SS.SEC='C';


--QUERY 2

SELECT SS.SEM, SS.SEC, S.GENDER, COUNT(S.GENDER) AS COUNT
FROM STUDENT S, SEMSEC SS, CLASS C
WHERE S.USN = C.USN AND SS.SSID = C.SSID
GROUP BY SS.SEM, SS.SEC, S.GENDER
ORDER BY SEM;

WHEN IA.FINALIA BETWEEN 50 AND 50 THEN 'OUTSTANDING'

END) AS CAT
FROM STUDENT S, SEMSEC SS, IAMARKS IA, COURSE SUB

SS.SSID = IA.SSID AND
SUB.SUBCODE = IA.SUBCODE AND
SUB.SEM = 8;
